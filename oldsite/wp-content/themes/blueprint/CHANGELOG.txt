** 	hooks added
	blueprint_theme_start		( after opening <body> tag )
	blueprint_banner_before		( before main banner section )
	blueprint_banner_after 		( after main banner section )
	blueprint_content_before	( before main content/sidebar section )
	blueprint_content_after		( after main content/sidebar section )
	blueprint_before_footer 	( before <footer> container )
	blueprint_theme_end			( before closing <body> tag )
	
	*** usage
